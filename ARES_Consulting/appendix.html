# ARES 로버 메인 애플리케이션

from machine import UART, Pin
import utime
from process_data import CommandProcessor
from hardware import robot
from pins import UART_TX_PIN, UART_RX_PIN

UART_ID = 0
UART_BAUDRATE = 9600
MAIN_LOOP_DELAY_MS = 10
RECEIVE_TIMEOUT_MS = 50
MAX_BUFFER_SIZE = 512


class AresRover:
    """ARES 로버 메인 애플리케이션 클래스"""

    def __init__(self):
        self.uart = UART(UART_ID, baudrate=UART_BAUDRATE,
                         tx=Pin(UART_TX_PIN), rx=Pin(UART_RX_PIN))
        self.processor = CommandProcessor()
        self.robot = robot
        self.is_running = False
        self.rx_buffer = ""

    def boot(self):
        """부팅 시퀀스 실행"""
        if self.uart.any():
            print("Cleaning UART buffer...")
            while self.uart.any():
                self.uart.read()
        self.rx_buffer = ""
        self._safe_stop_all_motors()
        self.robot.buzzer.boot_sound()
        self.robot.oled.fill(0)
        self.robot.oled.text("ARES READY", 0, 0)
        self.robot.oled.show()
        print("ARES Rover: Waiting for data...")
        self.is_running = True

    def _safe_stop_all_motors(self):
        self.robot.stop_all()

    def _read_uart_line(self):
        """UART에서 완전한 라인 읽기"""
        if not self.uart.any():
            return None
        while self.uart.any():
            chunk = self.uart.read()
            if chunk:
                self.rx_buffer += chunk.decode('utf-8', 'ignore')
            if len(self.rx_buffer) > MAX_BUFFER_SIZE:
                print("[UART] 버퍼 오버플로우, 초기화")
                self.rx_buffer = ""
                return None
            utime.sleep_ms(5)
        utime.sleep_ms(RECEIVE_TIMEOUT_MS)
        while self.uart.any():
            chunk = self.uart.read()
            if chunk:
                self.rx_buffer += chunk.decode('utf-8', 'ignore')
        if '\n' in self.rx_buffer:
            idx = self.rx_buffer.index('\n')
            complete_line = self.rx_buffer[:idx].strip()
            self.rx_buffer = self.rx_buffer[idx + 1:]
            return complete_line
        if self.rx_buffer:
            data = self.rx_buffer.strip()
            if data and not data.endswith(','):
                self.rx_buffer = ""
                return data
        return None

    def _is_status_message(self, data):
        return data.startswith('+')

    def _send_response(self, response):
        """UART로 응답 전송 (20바이트 청킹)"""
        data = response.encode('utf-8') + b'\n'
        CHUNK_SIZE = 20
        for i in range(0, len(data), CHUNK_SIZE):
            self.uart.write(data[i:i + CHUNK_SIZE])
            utime.sleep_ms(10)

    def _process_command(self, data):
        try:
            print(f"[CMD] {data}")
            result = self.processor.process(data)
            return f"{result}"
        except Exception as e:
            print(f"처리 오류: {e}")
            return "ERROR"

    def run(self):
        self.boot()
        while self.is_running:
            try:
                data = self._read_uart_line()
                if not data:
                    pass
                elif self._is_status_message(data):
                    print(f"BT 상태: {data}")
                else:
                    response = self._process_command(data)
                    self._send_response(response)
            except Exception as e:
                print(f"UART 오류: {e}")
            utime.sleep_ms(MAIN_LOOP_DELAY_MS)


if __name__ == "__main__":
    app = AresRover()
    app.run()
process_data.py — UART 명령 파싱 및 실행
# ARES 로버 명령 처리기

import utime
import time
from hardware import robot
from system_config import sys_config

PWM_MAX = 65535


class CommandProcessor:
    """UART 명령 파싱 및 실행 클래스"""

    def __init__(self):
        pass

    def _get_speed_pwm(self):
        spd = sys_config.get('max_speed') / 100.0
        return int(spd * PWM_MAX)

    def process(self, data):
        """명령 처리 및 결과 반환"""
        if data == "PING":          return self._handle_ping()
        elif data == "READY":       return "READY"
        elif data == "STOP_ALL":    return self._handle_stop_all()
        elif data == "GET_STATUS":  return self._handle_get_status()
        elif data == "GET_SYS":     return self._handle_get_sys()
        elif data == "GET_MODULES": return self._handle_get_modules()
        elif data.startswith("SYS_SET"):    return self._handle_sys_set(data)
        elif data.startswith("SET_PIN"):    return self._handle_set_pin(data)
        elif data.startswith("SET_MODULE"): return self._handle_set_module(data)
        elif data.startswith("BUZZER_ON"):  return self._handle_buzzer_on(data)
        elif data.startswith("SING"):       return self._handle_sing()
        elif data == "DISTANCE":
            return f"DIST:{robot.distance_sensor.get_distance()}" if robot.distance_sensor else "DIST:ERROR"
        elif data == "MAGNET":
            return f"MAG:{robot.mag_sensor.detect()}" if robot.mag_sensor else "MAG:ERROR"
        # 서보 이동 (시간 제한)
        elif data.startswith("SERVO_tFORWARD"): return self._handle_timed_wheel(data, "forward")
        elif data.startswith("SERVO_tBACKWARD"): return self._handle_timed_wheel(data, "backward")
        elif data.startswith("SERVO_tRIGHT"):   return self._handle_timed_wheel(data, "right")
        elif data.startswith("SERVO_tLEFT"):    return self._handle_timed_wheel(data, "left")
        # 레거시 호환
        elif data.startswith("tFORWARD"):  return self._handle_timed_wheel(data, "forward")
        elif data.startswith("tBACKWARD"): return self._handle_timed_wheel(data, "backward")
        elif data.startswith("tRIGHT"):    return self._handle_timed_wheel(data, "right")
        elif data.startswith("tLEFT"):     return self._handle_timed_wheel(data, "left")
        # 서보 이동 (연속)
        elif data == "SERVO_FORWARD":  return self._handle_continuous_wheel("forward")
        elif data == "SERVO_BACKWARD": return self._handle_continuous_wheel("backward")
        elif data == "SERVO_LEFT":     return self._handle_continuous_wheel("left")
        elif data == "SERVO_RIGHT":    return self._handle_continuous_wheel("right")
        elif data == "SERVO_STOP":
            robot.wheel.stop(); return 1
        # 레거시 호환
        elif data == "FORWARD":  return self._handle_continuous_wheel("forward")
        elif data == "BACKWARD": return self._handle_continuous_wheel("backward")
        elif data == "LEFT":     return self._handle_continuous_wheel("left")
        elif data == "RIGHT":    return self._handle_continuous_wheel("right")
        elif data == "STOP":
            robot.wheel.stop(); return 1
        # DC 모터
        elif data.startswith("DC_tFORWARD"):  return self._handle_timed_dcmotor_new(data, "forward")
        elif data.startswith("DC_tBACKWARD"): return self._handle_timed_dcmotor_new(data, "backward")
        elif data == "DC_FORWARD":  return self._handle_main_motor("forward")
        elif data == "DC_BACKWARD": return self._handle_main_motor("backward")
        elif data == "DC_STOP":
            robot.dcmotor.dc_stop(); return 1
        # 레거시 DC 모터
        elif data.startswith("DCMOTOR"):  return self._handle_dcmotor(data)
        elif data.startswith("tDCMOTOR"): return self._handle_timed_dcmotor(data)
        elif data == "MAIN_FORWARD":  return self._handle_main_motor("forward")
        elif data == "MAIN_BACKWARD": return self._handle_main_motor("backward")
        elif data == "MAIN_STOP":
            robot.dcmotor.dc_stop(); return 1
        # 캘리브레이션
        elif data.startswith("CALIB_START"): return self._handle_calib_start()
        elif data.startswith("CALIB_SET"):   return self._handle_calib_set(data)
        # LED
        elif data == "LED_PATTERN":
            robot.leds.swipe_effect(); return 1
        elif data.startswith("MAIN_LED_ON"):  return self._handle_main_led_on(data)
        elif data == "MAIN_LED_OFF":          return self._handle_main_led_off(data)
        elif data.startswith("LED_ON"):       return self._handle_led_on(data)
        elif data.startswith("LED_OFF"):      return self._handle_led_off(data)
        elif data.startswith("["):            return self._handle_led_pattern(data)
        # 디스플레이
        elif data == "CLEAR_DISPLAY":
            robot.oled.fill(0); robot.oled.show(); return 1
        elif data.startswith("MSG,"):         return self._handle_msg(data)
        # 전투
        elif data.startswith("GUN_FIRE"):
            if robot.gun: robot.gun.fire_once()
            return 1
        # 대기
        elif data.startswith("SLEEP"):        return self._handle_sleep(data)
        return 0

    def _handle_ping(self):
        if robot.oled:
            robot.oled.fill(0)
            robot.oled.text("CONNECTED!", 0, 0)
            robot.oled.text(str(sys_config.get('device_name')), 0, 16)
            robot.oled.show()
        return 1

    def _handle_stop_all(self):
        robot.stop_all()
        if robot.oled:
            robot.oled.fill(0)
            robot.oled.text("EMERGENCY STOP", 0, 0)
            robot.oled.show()
        return 1

    def _handle_get_status(self):
        s = robot.get_status_dict()
        return f"STATUS,{s['dist']},{s['mag']},{s['temp']},{s['mem']},{s['uptime']}"

    def _handle_get_sys(self):
        c = sys_config.config
        l_cal = c.get('left_calibration', 100)
        r_cal = c.get('right_calibration', 100)
        return f"SYS_VALUES,{c['max_speed']},{c['collision_dist']},{c['auto_stop']},{c['device_name']},{l_cal},{r_cal}"

    def _handle_get_modules(self):
        try:
            info = sys_config.get_module_info()
            result = "MODULES,"
            for module, data in info.items():
                status = "ON" if data['enabled'] else "OFF"
                result += f"{module}:{status},"
            return result.rstrip(',')
        except Exception as e:
            return "MODULES,ERROR"

    def _handle_sys_set(self, data):
        try:
            parts = data.split(',')
            if len(parts) >= 5:
                sys_config.save_config(parts[1], parts[2], parts[3], ",".join(parts[4:]))
                return 1
        except Exception as e:
            print(f"SYS_SET 오류: {e}")
        return 0

    def _handle_set_pin(self, data):
        try:
            parts = data.split(',')
            if len(parts) >= 3:
                if sys_config.set_pin(f"pin_{parts[1]}", int(parts[2])):
                    return f"PIN_SET,{parts[1]},{parts[2]}"
        except Exception as e:
            print(f"SET_PIN 오류: {e}")
        return "PIN_SET,ERROR"

    def _handle_set_module(self, data):
        try:
            parts = data.split(',')
            if len(parts) >= 3:
                if sys_config.enable_module(parts[1], int(parts[2])):
                    return f"MODULE_SET,{parts[1]},{'ON' if int(parts[2]) else 'OFF'}"
        except Exception as e:
            print(f"SET_MODULE 오류: {e}")
        return "MODULE_SET,ERROR"

    def _handle_buzzer_on(self, data):
        if not robot.buzzer: return 0
        try:
            argv = data.split(',')
            robot.buzzer.start(freq=int(float(argv[1])), duration=float(argv[2]), vol=50000)
            while robot.buzzer.is_playing:
                robot.buzzer.update()
                utime.sleep_ms(10)
            return 1
        except Exception as e:
            return 0

    def _handle_sing(self):
        if not robot.buzzer: return 0
        if robot.oled:
            robot.oled.fill(0); robot.oled.text("Singing...", 0, 0); robot.oled.show()
        robot.buzzer.halamadrid()
        return 1

    def _handle_timed_wheel(self, data, direction):
        if not robot.wheel: return 0
        try:
            sec = float(data.split(',')[1])
            spd = sys_config.get('max_speed') / 100.0
            getattr(robot.wheel, {'forward':'forward','backward':'backward',
                                   'right':'turn_right','left':'turn_left'}[direction])(speed=spd)
            time.sleep(sec)
            robot.wheel.stop()
            return 1
        except: return 0

    def _handle_continuous_wheel(self, direction):
        if not robot.wheel: return 0
        spd = sys_config.get('max_speed') / 100.0
        getattr(robot.wheel, {'forward':'forward','backward':'backward',
                               'right':'turn_right','left':'turn_left'}[direction])(speed=spd)
        return 1

    def _handle_timed_dcmotor_new(self, data, direction):
        if not robot.dcmotor: return 0
        try:
            sec = float(data.split(',')[1])
            pwm_val = self._get_speed_pwm()
            if direction == "forward": robot.dcmotor.dc_forward(pwm_val)
            else: robot.dcmotor.dc_backward(pwm_val)
            time.sleep(sec)
            robot.dcmotor.dc_stop()
            return 1
        except: return 0

    def _handle_main_motor(self, direction):
        if not robot.dcmotor: return 0
        pwm_val = self._get_speed_pwm()
        if direction == "forward": robot.dcmotor.dc_forward(pwm_val)
        else: robot.dcmotor.dc_backward(pwm_val)
        return 1

    def _handle_calib_start(self):
        if not robot.wheel or not robot.oled: return 0
        robot.oled.fill(0); robot.oled.text("Calibration...", 0, 0); robot.oled.show()
        robot.wheel.forward(speed=0.1)
        time.sleep(3)
        robot.wheel.stop()
        robot.oled.text("Done", 0, 16); robot.oled.show()
        return 1

    def _handle_calib_set(self, data):
        if not robot.wheel: return 0
        try:
            parts = data.split(',')
            sys_config.save_calibration(int(parts[1]), int(parts[2]))
            robot.wheel.update_factors(int(parts[1]), int(parts[2]))
            return 1
        except: return 0

    def _handle_main_led_on(self, data):
        if not robot.leds: return 0
        try:
            parts = data.split(',')
            robot.leds.main_led_on(float(parts[1]) if len(parts) > 1 else 1.0)
            return 1
        except: return 0

    def _handle_main_led_off(self, data):
        if not robot.leds: return 0
        robot.leds.main_led_off(); return 1

    def _handle_led_on(self, data):
        if not robot.leds: return 0
        try:
            parts = data.split(',')
            led_num = int(parts[1])
            brightness = max(0.0, min(1.0, float(parts[2]) if len(parts) > 2 else 1.0))
            if 0 <= led_num <= 4:
                robot.leds.leds[led_num].duty_u16(int(65535 * brightness))
            return 1
        except: return 0

    def _handle_led_off(self, data):
        if not robot.leds: return 0
        try:
            parts = data.split(',')
            led_num = parts[1] if len(parts) > 1 else '0'
            if led_num == 'ALL': robot.leds.leds_off()
            else:
                num = int(led_num)
                if 0 <= num <= 4: robot.leds.leds[num].duty_u16(0)
            return 1
        except: return 0

    def _handle_led_pattern(self, data):
        if not robot.leds: return 0
        try:
            values = data.strip('[]').split()
            for i, val in enumerate(values):
                if i >= 5: break
                robot.leds.leds[i].duty_u16(int(65535 * max(0.0, min(1.0, float(val)))))
            return 1
        except: return 0

    def _handle_msg(self, data):
        if not robot.oled: return 0
        try:
            msg = str(data[data.find(',') + 1:]) or "Hello"
            robot.oled.fill(0)
            y_pos = 0
            while msg and y_pos < 64:
                robot.oled.text(msg[:16], 0, y_pos)
                msg = msg[16:]; y_pos += 10
            robot.oled.show()
            return 1
        except: return 0

    def _handle_sleep(self, data):
        try:
            time.sleep(float(data.split(',')[1])); return 1
        except: return 0

    def _handle_dcmotor(self, data):
        if not robot.dcmotor: return 0
        try:
            parts = data.split(',')
            direction, speed = parts[1], max(0, min(100, int(float(parts[2]))))
            pwm_val = int(speed * PWM_MAX / 100)
            if direction == "FORWARD": robot.dcmotor.dc_forward(pwm_val)
            elif direction == "BACKWARD": robot.dcmotor.dc_backward(pwm_val)
            elif direction == "STOP": robot.dcmotor.dc_stop()
            return 1
        except: return 0

    def _handle_timed_dcmotor(self, data):
        if not robot.dcmotor: return 0
        try:
            parts = data.split(',')
            direction, sec = parts[1], float(parts[2])
            pwm_val = self._get_speed_pwm()
            if direction == "FORWARD": robot.dcmotor.dc_forward(pwm_val)
            elif direction == "BACKWARD": robot.dcmotor.dc_backward(pwm_val)
            time.sleep(sec); robot.dcmotor.dc_stop()
            return 1
        except: return 0

# ARES 로버 하드웨어 싱글톤

from machine import I2C, Pin, UART
from system_config import sys_config
import time, gc, machine


class RobotHardware:
    """하드웨어 싱글톤 클래스"""
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super(RobotHardware, cls).__new__(cls)
            cls._instance._init_hardware()
        return cls._instance

    def _init_hardware(self):
        self.temp_sensor = machine.ADC(4)
        self.start_time = time.ticks_ms()
        self.dcmotor = self.oled = self.i2c = self.buzzer = None
        self.distance_sensor = self.wheel = self.mag_sensor = None
        self.leds = self.gun = None

        if sys_config.is_module_enabled('dcmotor'):
            try:
                from dcmotor import KSDCMotor
                self.dcmotor = KSDCMotor(); self.dcmotor.dc_stop()
                print("[HW] DC 모터 초기화 완료")
            except Exception as e:
                print(f"[HW] DC 모터 초기화 실패: {e}")

        if sys_config.is_module_enabled('oled'):
            try:
                from ssd1306 import SSD1306_I2C
                i2c_sda = sys_config.get('pin_i2c_sda')
                i2c_scl = sys_config.get('pin_i2c_scl')
                self.i2c = I2C(0, scl=Pin(i2c_scl), sda=Pin(i2c_sda), freq=400000)
                self.oled = SSD1306_I2C(128, 64, scl_pin=i2c_scl, sda_pin=i2c_sda)
                print("[HW] OLED 디스플레이 초기화 완료")
            except Exception as e:
                print(f"[HW] OLED 초기화 실패: {e}"); self.oled = None

        if sys_config.is_module_enabled('buzzer'):
            try:
                from buzzer import KSBuzzer
                self.buzzer = KSBuzzer(); print("[HW] 부저 초기화 완료")
            except Exception as e:
                print(f"[HW] 부저 초기화 실패: {e}")

        if sys_config.is_module_enabled('distance'):
            try:
                from ultrasonic import KSDistance
                self.distance_sensor = KSDistance(); print("[HW] 거리 센서 초기화 완료")
            except Exception as e:
                print(f"[HW] 거리 센서 초기화 실패: {e}")

        if sys_config.is_module_enabled('wheel'):
            try:
                from wheel import KSWheel
                self.wheel = KSWheel(); self.wheel.stop()
                print("[HW] 서보 모터 초기화 완료")
            except Exception as e:
                print(f"[HW] 서보 모터 초기화 실패: {e}")

        if sys_config.is_module_enabled('magsensor'):
            try:
                from magsensor import KSMagSensor
                self.mag_sensor = KSMagSensor(); print("[HW] 자기장 센서 초기화 완료")
            except Exception as e:
                print(f"[HW] 자기장 센서 초기화 실패: {e}")

        if sys_config.is_module_enabled('leds'):
            try:
                from leds import KSLeds
                self.leds = KSLeds(); print("[HW] LED 초기화 완료")
            except Exception as e:
                print(f"[HW] LED 초기화 실패: {e}")

        if sys_config.is_module_enabled('gun'):
            try:
                from gun import KSGun
                self.gun = KSGun(); print("[HW] 총 초기화 완료")
            except Exception as e:
                print(f"[HW] 총 초기화 실패: {e}")

        self._sync_bt_name()

    def _sync_bt_name(self):
        try:
            from pins import UART_TX_PIN, UART_RX_PIN
            target_name = sys_config.get('device_name')
            uart = UART(0, baudrate=9600, tx=Pin(UART_TX_PIN), rx=Pin(UART_RX_PIN))
            uart.write(f"AT+NAME{target_name}".encode('utf-8'))
            time.sleep(0.1)
            print(f"[HW] BT 이름 설정: {target_name}")
        except Exception as e:
            print(f"[HW] BT 이름 동기화 오류: {e}")

    def get_temperature(self):

    def __init__(self, buff, w, h, oled):
        self.w = w; self.h = h
        self.buff = buff; self.oled = oled
        # 수평 → 수직(MONO_VLSB) 포맷 변환
        vertical_buff = bytearray(self.w * self.h // 8)
        bytes_per_row  = self.w // 8
        for row in range(self.h):
            for byte_col in range(bytes_per_row):
                byte = buff[row * bytes_per_row + byte_col]
                for bit in range(8):
                    if (byte >> (7 - bit)) & 1:
                        px = byte_col * 8 + bit
                        py = row
                        vi = px + (py // 8) * self.w
                        vertical_buff[vi] |= (1 << (py % 8))
        self.fb = framebuf.FrameBuffer(vertical_buff, self.w, self.h, framebuf.MONO_VLSB)

    def blit(self, x, y):
        for i in range(self.w):

(function() {
    'use strict';

    const CONFIG = {
        device_name: '', max_speed: 0, collision_distance: 0,
        auto_stop: false, left_calibration: 0, right_calibration: 0,
        connection_timeout: 20000
    };

    let lastUpdateTime = 0, isConnected = false;
    const DOM = {};

    function initDOMReferences() {
        DOM.connectionStatus = document.getElementById('connection-status');
        DOM.bleStatus        = document.getElementById('ble-status');
        DOM.distanceValue    = document.getElementById('distance-value');
        DOM.distBar          = document.getElementById('dist-bar');
        DOM.magneticValue    = document.getElementById('magnetic-value');
        DOM.deviceName       = document.getElementById('device-name');
        DOM.maxSpeed         = document.getElementById('max-speed');
        DOM.calibLeft        = document.getElementById('calib-left');
        DOM.calibRight       = document.getElementById('calib-right');
        // ... 기타 DOM 참조
    }

    function updateConnectionUI(connected) {
        isConnected = connected;
        if (DOM.connectionStatus)
            DOM.connectionStatus.classList.toggle('connected', connected);
        if (DOM.bleStatus)
            DOM.bleStatus.textContent = connected ? '연결됨' : '연결 끊김';
    }

    function updateSensorData(rawData) {
        // "STATUS,dist,mag,temp,mem,uptime" 파싱
        const parts = rawData.split(',');
        if (parts[0] !== 'STATUS' || parts.length < 6) return;
        const dist = parseFloat(parts[1]);
        if (DOM.distanceValue) DOM.distanceValue.textContent = `${dist.toFixed(1)} cm`;
        if (DOM.distBar) DOM.distBar.style.width = `${Math.min(dist / 4, 100)}%`;
        if (DOM.magneticValue) {
            DOM.magneticValue.textContent = parts[2] === '1' ? '감지됨' : '없음';
            DOM.magneticValue.style.color = parts[2] === '1'
                ? 'var(--success-color)' : 'var(--text-dim)';
        }
        lastUpdateTime = Date.now();
    }

    function sendCommandToParent(cmd) {
        window.parent.postMessage({ type: 'command', data: cmd }, '*');
    }

    function initCommandButtons() {
        document.querySelectorAll('.action-btn[data-cmd]').forEach(btn => {
            btn.addEventListener('click', () => {
                const cmdMap = {
                    forward: 'SERVO_FORWARD', stop: 'STOP_ALL',
                    led: 'LED_PATTERN', left: 'SERVO_LEFT',
                    right: 'SERVO_RIGHT', sound: 'SING'
                };
                const cmd = cmdMap[btn.dataset.cmd];
                if (cmd) sendCommandToParent(cmd);
            });
        });
    }

    function initSettingsButtons() {
        document.getElementById('btn-save-system')?.addEventListener('click', () => {
            const name  = DOM.deviceName?.value || 'ARES';
            const speed = DOM.maxSpeed?.value   || '80';
            const dist  = document.getElementById('collision-distance')?.value || '10';
            const stop  = document.getElementById('auto-stop')?.value === 'true' ? 1 : 0;
            sendCommandToParent(`SYS_SET,${speed},${dist},${stop},${name}`);
        });

        document.getElementById('btn-calib-start')?.addEventListener('click', () =>
            sendCommandToParent('CALIB_START'));

        document.getElementById('btn-calib-apply')?.addEventListener('click', () => {
            const l = DOM.calibLeft?.value  || '100';
            const r = DOM.calibRight?.value || '100';
            sendCommandToParent(`CALIB_SET,${l},${r}`);
        });
    }

    function handleMessage(event) {
        const data = event.data;
        if (!data?.type) return;
        switch (data.type) {
            case 'connection_status':
                updateConnectionUI(data.connected);
                setDashboardInteractive(data.connected);
                if (data.connected) {
                    setTimeout(() => sendCommandToParent('GET_SYS'),  500);
                    setTimeout(() => sendCommandToParent('GET_STATUS'), 700);
                } else {
                    resetDashboardValuesToZero();
                }
                break;
            case 'status_update': updateSensorData(data.data); break;
            case 'sys_values':
                CONFIG.max_speed = parseInt(data.max_speed, 10) || 0;
                CONFIG.collision_distance = parseInt(data.collision_dist, 10) || 0;
                CONFIG.auto_stop = data.auto_stop == 1;
                CONFIG.device_name = data.device_name || '';
                updateSettingsUI();
                break;
        }
    }

    function init() {
        initDOMReferences();
        initTabs();
        initCommandButtons();
        initSettingsButtons();
        resetDashboardValuesToZero();
        setDashboardInteractive(false);
        window.addEventListener('message', handleMessage);
        startConnectionMonitor();
    }

    if (document.readyState === 'loading')
        document.addEventListener('DOMContentLoaded', init);
    else
        init();
})();

// 로그 관리

import { elements } from './elements.js';

const MAX_COMPACT_LINES = 3;
const MAX_ENTRIES = 500;
const entries = [];

function isExpanded() {
    return elements.logContainer?.classList.contains('expanded');
}

function escapeHtml(str) {
    return String(str)
        .replaceAll('&', '&').replaceAll('<', '<')
        .replaceAll('>', '>').replaceAll('"', '"');
}

function render() {
    const expanded = isExpanded();
    const visible = expanded
        ? entries
        : entries.filter(e => !e.verbose).slice(-MAX_COMPACT_LINES);

    elements.logContent.innerHTML = '';
    for (const entry of visible) {
        const el = document.createElement('div');
        el.className = `log-entry log-${entry.type}`;
        el.innerHTML = `
            ${entry.timestamp}
            ${escapeHtml(entry.message)}
            ${expanded && entry.detail
                ? `
${escapeHtml(entry.detail)}
`
                : ''}`;
        elements.logContent.appendChild(el);
    }
    if (expanded) elements.logContent.scrollTop = elements.logContent.scrollHeight;
}

export const Logger = {
    add(message, type = 'info', options = {}) {
        entries.push({
            timestamp: new Date().toLocaleTimeString(),
            message, type,
            verbose: !!options.verbose,
            detail:  options.detail || ''
        });
        if (entries.length > MAX_ENTRIES) entries.shift();
        render();
    },
    clear()   { entries.length = 0; elements.logContent.innerHTML = ''; },
    refresh() { render(); }
};
bluetooth.js — Web Bluetooth API 통신 관리
// 블루투스 매니저

import { state, DEBUG } from './state.js';
import { elements } from './elements.js';
import { Logger } from './logger.js';
import { BLUETOOTH_CONFIG, STATUS_COLORS } from './constants.js';

let receiveBuffer = '';

export const BluetoothManager = {
    async connect() {
        if (state.isConnecting) { Logger.add('[경고] 이미 연결 시도 중', 'error'); return; }
        state.isConnecting = true;
        elements.connectButton.disabled = true;
        this.updateStatus('아레스 탐색 중...', STATUS_COLORS.ORANGE);
        try {
            Logger.add('[BLE] 장치 검색 중...', 'info');
            state.bluetoothDevice = await navigator.bluetooth.requestDevice({
                filters: [{ name: 'PicoBLE' }, { name: 'HMSoft' }, { name: 'BT05' }],
                optionalServices: [BLUETOOTH_CONFIG.UART_SERVICE_UUID]
            });
            state.bluetoothDevice.addEventListener(
                'gattserverdisconnected', this.onDeviceDisconnected.bind(this));
            state.bluetoothServer = await state.bluetoothDevice.gatt.connect();
            await this.delay(2000);
            state.uartService = await state.bluetoothServer.getPrimaryService(
                BLUETOOTH_CONFIG.UART_SERVICE_UUID);
            state.characteristic = await state.uartService.getCharacteristic(
                BLUETOOTH_CONFIG.UART_CHARACTERISTIC_UUID);
            try {
                await state.characteristic.startNotifications();
                state.characteristic.addEventListener(
                    'characteristicvaluechanged', this.handleRxData.bind(this));
                state.notificationsEnabled = true;
                Logger.add('[BLE] 알림 모드 활성화', 'info');
            } catch {
                Logger.add('[BLE] 폴링 모드로 전환', 'info');
                this.startPeriodicReads();
            }
            this.updateConnectionStatus(true);
            Logger.add(`[연결] ${state.bluetoothDevice.name || 'Unknown'} 연결 완료`, 'success');
        } catch (error) {
            Logger.add(`[오류] 연결 실패: ${error.message}`, 'error');
            await this.cleanup();
            this.updateStatus(`❌ 연결 실패: ${error.message}`, STATUS_COLORS.RED);
        }
        state.isConnecting = false;
        elements.connectButton.disabled = false;
    },

    async disconnect() {
        try {
            if (state.characteristic && state.notificationsEnabled) {
                await state.characteristic.stopNotifications();
            }
            if (state.bluetoothDevice?.gatt.connected)
                await state.bluetoothDevice.gatt.disconnect();
            await this.cleanup();
            this.updateConnectionStatus(false);
            Logger.add('[연결] 해제 완료', 'info');
        } catch (error) {
            Logger.add(`[오류] 연결 해제 실패: ${error.message}`, 'error');
        }
    },

    async cleanup() {
        receiveBuffer = '';
        if (state.readIntervalId) { clearInterval(state.readIntervalId); state.readIntervalId = null; }
        state.characteristic = state.uartService = state.bluetoothServer = null;
        state.bluetoothDevice = null;
        state.notificationsEnabled = false;
        if (state.pendingTimeout) {
            clearTimeout(state.pendingTimeout);
            state.pendingResolve = state.pendingReject = state.pendingTimeout = null;
        }
    },

    handleRxData(event) {
        const chunk = new TextDecoder('utf-8').decode(event.target.value);
        receiveBuffer += chunk;
        let idx;
        while ((idx = receiveBuffer.indexOf('\n')) !== -1) {
            const msg = receiveBuffer.substring(0, idx).trim();
            receiveBuffer = receiveBuffer.substring(idx + 1);
            if (msg) this.processReceivedData(msg);
        }
        if (receiveBuffer.length > 1024) {
            const data = receiveBuffer.trim(); receiveBuffer = '';
            if (data) this.processReceivedData(data);
        }
    },

    processReceivedData(data) {
        if (!data) return;
        const iframe = document.getElementById('dashboardFrame');
        if (data.startsWith('STATUS,')) {
            iframe?.contentWindow?.postMessage({ type: 'status_update', data }, '*');
            this._resolvePromise(data); return;
        }
        if (data.startsWith('SYS_VALUES,')) { this._handleSysValues(data); return; }
        if (data.startsWith('CALIB_VALUES,')) { this._handleCalibValues(data); return; }
        if (DEBUG) Logger.add(`[수신] ${data}`, 'info');
        this._resolvePromise(data);
        this._updateBlocklyVariable(data);
    },

    _resolvePromise(data) {
        if (state.pendingResolve) {
            if (state.pendingTimeout) clearTimeout(state.pendingTimeout);
            const resolve = state.pendingResolve;
            state.pendingResolve = state.pendingReject = state.pendingTimeout = null;
            resolve(data);
        }
    },

    _updateBlocklyVariable(data) {
        const dm = data.match(/DIST[:\s]*([\d.]+)/i);
        if (dm) state.variables['_last_distance'] = dm[1];
        const mm = data.match(/MAG[:\s]*([\d]+)/i);
        if (mm) state.variables['_last_magnetic'] = mm[1];
    },

    _handleSysValues(data) {
        const parts = data.split(',');
        const iframe = document.getElementById('dashboardFrame');
        if (iframe?.contentWindow) {
            iframe.contentWindow.postMessage({
                type: 'sys_values',
                max_speed: parts[1], collision_dist: parts[2],
                auto_stop: parts[3], device_name: parts.slice(4, -2).join(','),
                left_calibration: parts.at(-2), right_calibration: parts.at(-1),
                connection_timeout: BLUETOOTH_CONFIG.RESPONSE_TIMEOUT
            }, '*');
        }
        this._resolvePromise(data);
        Logger.add('[수신] 시스템 설정값', 'success');
    },

    _handleCalibValues(data) {
        const parts = data.split(',');
        document.getElementById('dashboardFrame')?.contentWindow?.postMessage(
            { type: 'calib_values', left: parts[1], right: parts[2] }, '*');
        this._resolvePromise(data);
    },

    startPeriodicReads() {
        if (!state.readIntervalId) {
            state.readIntervalId = setInterval(
                () => this.readCharacteristic(), BLUETOOTH_CONFIG.READ_INTERVAL);
        }
    },

    async sendData(data, waitForResponse = false) {
        if (!state.characteristic) throw new Error('BLE 장치에 연결되어 있지 않습니다.');
        Logger.add(`[전송] ${data}`, 'send');
        const encoded = new TextEncoder().encode(data + '\n');
        state.lastCommand = data;
        try {
            for (let i = 0; i < encoded.length; i += BLUETOOTH_CONFIG.MAX_CHUNK_SIZE) {
                await state.characteristic.writeValueWithResponse(
                    encoded.slice(i, i + BLUETOOTH_CONFIG.MAX_CHUNK_SIZE));
                await this.delay(BLUETOOTH_CONFIG.CHUNK_DELAY);
            }
        } catch (error) {
            if (DEBUG) Logger.add(`전송 오류 무시됨: ${error.message}`, 'warning');
        }
        if (!waitForResponse) return 'OK';
        return new Promise((resolve, reject) => {
            state.pendingResolve = resolve;
            state.pendingReject  = reject;
            state.pendingTimeout = setTimeout(() => {
                state.pendingResolve = state.pendingReject = state.pendingTimeout = null;
                reject(new Error('응답 시간 초과'));
            }, BLUETOOTH_CONFIG.RESPONSE_TIMEOUT);
        });
    },

    updateConnectionStatus(connected) {
        elements.connectButton.disabled    = connected || state.isConnecting;
        elements.disconnectButton.disabled = !connected;
        const iframe = document.getElementById('dashboardFrame');
        iframe?.contentWindow?.postMessage({ type: 'connection_status', connected }, '*');
        if (connected) {
            elements.status.textContent = '✅ 아레스 연결됨';
            elements.status.style.color = STATUS_COLORS.GREEN;
            if (state.bluetoothDevice)
                elements.deviceInfo.textContent = `장치: ${state.bluetoothDevice.name || 'Unknown'}`;
        } else {
            elements.status.textContent = state.isConnecting ? '연결 중...' : '❌ 연결 끊김';
            elements.status.style.color = state.isConnecting ? STATUS_COLORS.ORANGE : STATUS_COLORS.RED;
            elements.deviceInfo.textContent = '';
        }
    },

    updateStatus(msg, color) {
        elements.status.textContent = msg;
        elements.status.style.color = color;
    },

    delay(ms) { return new Promise(r => setTimeout(r, ms)); }
};

import { BluetoothManager } from './bluetooth.js';
import { state, DEBUG } from './state.js';
import { elements } from './elements.js';
import { Logger } from './logger.js';
import { BLUETOOTH_CONFIG, STATUS_COLORS } from './constants.js';

export const CommandExecutor = {
    evaluateValueBlock(block) {
        if (!block) return '0';
        switch (block.type) {
            case 'math_number':    return block.getFieldValue('NUM') || '0';
            case 'text':           return block.getFieldValue('TEXT') || '';
            case 'variables_get': {
                const name = block.workspace.getVariableById(block.getFieldValue('VAR'))?.name || '';
                return state.variables[name] || '0';
            }
            case 'math_arithmetic': {
                const op = block.getFieldValue('OP');
                const a = parseFloat(this.evaluateValueBlock(block.getInputTargetBlock('A')));
                const b = parseFloat(this.evaluateValueBlock(block.getInputTargetBlock('B')));
                switch (op) {
                    case 'ADD':      return (a + b).toString();
                    case 'MINUS':    return (a - b).toString();
                    case 'MULTIPLY': return (a * b).toString();
                    case 'DIVIDE':   return (b !== 0 ? a / b : 0).toString();
                    default:         return '0';
                }
            }
            case 'logic_compare': {
                const op = block.getFieldValue('OP');
                const a = parseFloat(this.evaluateValueBlock(block.getInputTargetBlock('A')));
                const b = parseFloat(this.evaluateValueBlock(block.getInputTargetBlock('B')));
                const map = { EQ:'===', NEQ:'!==', LT:'<', LTE:'<=', GT:'>', GTE:'>=' };
                return eval(`${a} ${map[op]} ${b}`) ? 'true' : 'false';
            }
            case 'logic_boolean':
                return block.getFieldValue('BOOL') === 'TRUE' ? 'true' : 'false';
            case 'math_random_int': {
                const from = parseInt(this.evaluateValueBlock(block.getInputTargetBlock('FROM'))) || 0;
                const to   = parseInt(this.evaluateValueBlock(block.getInputTargetBlock('TO')))   || 100;
                const mn = Math.min(from, to), mx = Math.max(from, to);
                return (Math.floor(Math.random() * (mx - mn + 1)) + mn).toString();
            }
            default: return '0';
        }
    },

    generateCommand(block) {
        switch (block.type) {
            case 'set_lamp': {
                const lamps = [0,1,2,3,4].map(i =>
                    parseFloat(this.evaluateValueBlock(block.getInputTargetBlock(`LAMP${i}`)) || '0').toFixed(1));
                return `[${lamps.join(' ')}]`;
            }
            case 'led_on': {
                const n  = Math.max(0, Math.min(4,
                    parseInt(this.evaluateValueBlock(block.getInputTargetBlock('LED_NUM')), 10) - 1));
                const br = this.evaluateValueBlock(block.getInputTargetBlock('BRIGHTNESS')) || '1';
                return `LED_ON,${n},${br}`;
            }
            case 'led_off': {
                const s = block.getFieldValue('LED_NUM') || '1';
                if (s === 'ALL') return 'LED_OFF,ALL';
                return `LED_OFF,${Math.max(0, Math.min(4, parseInt(s) - 1))}`;
            }
            case 'main_led_on':   return `MAIN_LED_ON,${this.evaluateValueBlock(block.getInputTargetBlock('BRIGHTNESS')) || '1'}`;
            case 'main_led_off':  return 'MAIN_LED_OFF';
            case 'send_message':  return `MSG,${this.evaluateValueBlock(block.getInputTargetBlock('Msg')) || 'Hello'}`;
            case 'clear_display': return 'CLEAR_DISPLAY';
            case 'buzzer_on': {
                const freq = Math.trunc(parseFloat(this.evaluateValueBlock(block.getInputTargetBlock('FREQ')) || '262'));
                const dur  = this.evaluateValueBlock(block.getInputTargetBlock('DURATION')) || '1';
                return `BUZZER_ON,${freq},${dur}`;
            }
            case 'gun_fire':              return 'GUN_FIRE';
            case 'timed_forward':         return `SERVO_tFORWARD,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS')) || '0'}`;
            case 'timed_backward':        return `SERVO_tBACKWARD,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS')) || '0'}`;
            case 'timed_right':           return `SERVO_tRIGHT,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS'))   || '0'}`;
            case 'timed_left':            return `SERVO_tLEFT,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS'))    || '0'}`;
            case 'move_forward':          return 'SERVO_FORWARD';
            case 'move_backward':         return 'SERVO_BACKWARD';
            case 'turn_left':             return 'SERVO_LEFT';
            case 'turn_right':            return 'SERVO_RIGHT';
            case 'stop_moving':           return 'SERVO_STOP';
            case 'main_motor_forward_timed':  return `DC_tFORWARD,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS')) || '1'}`;
            case 'main_motor_backward_timed': return `DC_tBACKWARD,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS')) || '1'}`;
            case 'main_motor_forward':    return 'DC_FORWARD';
            case 'main_motor_backward':   return 'DC_BACKWARD';
            case 'main_motor_stop':       return 'DC_STOP';
            case 'time_sleep':            return `SLEEP,${this.evaluateValueBlock(block.getInputTargetBlock('SECONDS')) || '0'}`;
            case 'pico_check_device':     return 'PING';
            case 'check_distance':        return 'DISTANCE';
            case 'check_magnetic':        return 'MAGNET';
            default: return null;
        }
    },

    async processBlock(block) {
        if (!block || !state.isExecuting) return;
        const command = this.generateCommand(block);
        if (command) await this.sendCommand(command);
        await this.handleLogicBlock(block);
        await this.processBlock(block.getNextBlock());
    },

    async sendCommand(command) {
        if (!state.isExecuting) { Logger.add('[중단] 실행 중단', 'warning'); return; }
        BluetoothManager.updateStatus('명령 실행 중...', STATUS_COLORS.ORANGE);
        try {
            await BluetoothManager.sendData(command, true);
        } catch (error) {
            if (error.message.includes('시간 초과')) Logger.add(`[경고] 응답 초과: ${command}`, 'warning');
            else { Logger.add(`[오류] ${command}: ${error.message}`, 'error'); throw error; }
        }
        await new Promise(r => setTimeout(r, 100));
    },

    async handleLogicBlock(block) {
        if (block.type === 'variables_set') {
            const name = block.workspace.getVariableById(block.getFieldValue('VAR'))?.name || '';
            state.variables[name] = this.evaluateValueBlock(block.getInputTargetBlock('VALUE'));
        } else if (block.type === 'controls_if') {
            const cond = this.evaluateValueBlock(block.getInputTargetBlock('IF0')) === 'true';
            await this.processBlock(block.getInputTargetBlock(cond ? 'DO0' : 'ELSE'));
        } else if (block.type === 'controls_whileUntil') {
            const mode = block.getFieldValue('MODE');
            let cond = this.evaluateValueBlock(block.getInputTargetBlock('BOOL')) === 'true';
            let n = 0;
            while ((mode === 'WHILE' ? cond : !cond) && n++ < 100 && state.isExecuting) {
                await this.processBlock(block.getInputTargetBlock('DO'));
                cond = this.evaluateValueBlock(block.getInputTargetBlock('BOOL')) === 'true';
            }
        } else if (block.type === 'controls_repeat_ext') {
            const times = Math.min(parseInt(this.evaluateValueBlock(block.getInputTargetBlock('TIMES')) || '0'), 100);
            for (let i = 0; i < times && state.isExecuting; i++)
                await this.processBlock(block.getInputTargetBlock('DO'));
        }
    },

    async executeWorkspace(workspace) {
        state.isExecuting = true;
        elements.runButton.disabled = true;
        BluetoothManager.updateStatus('프로그램 실행 중...', STATUS_COLORS.ORANGE);
        Logger.add('[실행] 프로그램 시작', 'info');
        try {
            for (const block of workspace.getTopBlocks(true)) {
                if (!state.isExecuting) break;
                if (block.type.startsWith('procedures_def')) continue;
                await this.processBlock(block);
            }
            if (state.isExecuting) {
                BluetoothManager.updateStatus('✅ 실행 완료!', STATUS_COLORS.GREEN);
                Logger.add('[실행] 완료', 'info');
            }
        } catch (error) {
            BluetoothManager.updateStatus(`❌ 실행 실패: ${error.message}`, STATUS_COLORS.RED);
            Logger.add(`[오류] ${error.message}`, 'error');
        }
        state.isExecuting = false;
        elements.runButton.disabled = !(state.bluetoothDevice?.gatt?.connected && state.characteristic);
    }
};
